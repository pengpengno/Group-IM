// UserViewModelTest.kt
class UserViewModelTest {
    
    private lateinit var viewModel: UserViewModel
    private lateinit var mockUserRepository: UserRepository
    private lateinit var mockLoginStateManager: LoginStateManager
    
    @BeforeTest
    fun setup() {
        mockUserRepository = mock(classOf<UserRepository>())
        mockLoginStateManager = mock(classOf<LoginStateManager>())
        viewModel = UserViewModel(mockUserRepository, mockLoginStateManager)
    }
    
    @Test
    fun `登录成功时应该更新状态并保存凭据`() = runTest {
        // Given
        val username = "testuser"
        val password = "password123"
        val expectedUserInfo = createTestUserInfo()
        
        coEvery { 
            mockUserRepository.login(username, password) 
        } returns Result.success(expectedUserInfo)
        
        // When
        viewModel.login(username, password)
        
        // Then
        verify {
            mockUserRepository.updateToAuthenticating()
            mockUserRepository.updateToAuthenticated(expectedUserInfo)
        }
        
        // 验证凭据保存
        verify {
            mockUserRepository.saveCredentials(expectedUserInfo)
        }
    }
    
    @Test
    fun `登录失败时应该更新失败状态`() = runTest {
        // Given
        val username = "testuser"
        val password = "wrongpassword"
        val errorMessage = "Invalid credentials"
        
        coEvery { 
            mockUserRepository.login(username, password) 
        } returns Result.failure(UnauthorizedException(errorMessage))
        
        // When
        viewModel.login(username, password)
        
        // Then
        verify {
            mockUserRepository.updateToAuthenticating()
            mockUserRepository.updateToAuthenticationFailed(
                errorMessage, 
                isNetworkError = false
            )
        }
    }
    
    @Test
    fun `网络错误时应该标记为网络错误`() = runTest {
        // Given
        val exception = ConnectException("Network unavailable")
        
        coEvery { 
            mockUserRepository.login(any(), any()) 
        } returns Result.failure(exception)
        
        // When
        viewModel.login("user", "pass")
        
        // Then
        verify {
            mockUserRepository.updateToAuthenticationFailed(
                any(),
                isNetworkError = true
            )
        }
    }
}
```

#### 4.2.1 状态管理测试

// UserRepositoryTest.kt
class UserRepositoryTest {
    
    private lateinit var repository: UserRepository
    private lateinit var mockDatabase: AppDatabase
    
    @BeforeTest
    fun setup() {
        mockDatabase = mock(classOf<AppDatabase>())
        repository = UserRepository(mockDatabase)
    }
    
    @Test
    fun `状态转换应该按预期工作`() = runTest {
        // 测试各种状态转换
        repository.updateToChecking()
        assertEquals(LoginState.Checking, repository.userState.value)
        
        repository.updateToAuthenticating()
        assertEquals(LoginState.Authenticating, repository.userState.value)
        
        val userInfo = createTestUserInfo()
        repository.updateToAuthenticated(userInfo)
        val authenticatedState = repository.userState.value as LoginState.Authenticated
        assertEquals(userInfo, authenticatedState.userInfo)
    }
    
    @Test
    fun `并发状态更新应该安全`() = runTest {
        // 测试并发状态更新的安全性
        val jobs = List(100) {
            launch {
                repository.updateToAuthenticated(createTestUserInfo())
            }
        }
        
        jobs.joinAll()
        assertTrue(repository.userState.value is LoginState.Authenticated)
    }
}
```

### 4.3 LoginApi测试

#### 4.3.1 网络请求测试
```kotlin
// LoginApiTest.kt
class LoginApiTest {
    
    private lateinit var mockHttpClient: HttpClient
    
    @BeforeTest
    fun setup() {
        mockHttpClient = mock(classOf<HttpClient>())
    }
    
    @Test
    fun `成功登录应该返回用户信息`() = runTest {
        // Given
        val expectedResponse = createTestUserInfo()
        coEvery {
            mockHttpClient.post<LoginRequest, UserInfo>(any(), any())
        } returns expectedResponse
        
        // When
        val result = LoginApi.login("user", "pass", "")
        
        // Then
        assertEquals(expectedResponse, result)
    }
    
    @Test
    fun `401错误应该抛出认证异常`() = runTest {
        // Given
        coEvery {
            mockHttpClient.post<LoginRequest, UserInfo>(any(), any())
        } throws ClientRequestException(HttpStatusCode.Unauthorized, "Unauthorized")
        
        // When & Then
        assertFailsWith<UnauthorizedException> {
            LoginApi.login("user", "wrongpass", "")
        }
    }
    
    @Test
    fun `网络超时应该抛出连接异常`() = runTest {
        // Given
        coEvery {
            mockHttpClient.post<LoginRequest, UserInfo>(any(), any())
        } throws ConnectTimeoutException("Connection timeout")
        
        // When & Then
        val exception = assertFailsWith<ConnectException> {
            LoginApi.login("user", "pass", "")
        }
        assertTrue(exception.message?.contains("timeout") == true)
    }
}
```
